    private void calcular() {
        
        Calculadora calculadora = new Calculadora();
        
        double numero1 = Double.parseDouble(jtNumero1.getText());
        double numero2 = Double.parseDouble(jtNumero2.getText());
        String opcao = bgOpcao.getSelection().getActionCommand();
        
        JOptionPane.showMessageDialog(
                this, 
                calculadora.verificarOpcao(opcao, numero1, numero2));
    }
    private void limpar() {
        jtNumero1.setText("");
        jtNumero2.setText("");
        jtNumero1.grabFocus();
    }
    
    private void jbCalcularActionPerformed(java.awt.event.ActionEvent evt) {                                           
        // TODO add your handling code here:
        calcular();
        limpar();
    }

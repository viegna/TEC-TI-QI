    private void cadastrar() {
        Pessoa pessoa = new Pessoa();
        pessoa.setNome(jtNome.getText());
        
        if(jrMasculino.isSelected()) {
            pessoa.setSexo(jrMasculino.getText());
        } else if(jrFeminino.isSelected()) {
            pessoa.setSexo(jrFeminino.getText());
        } else {
            pessoa.setSexo("Outros");
        }//fecha else
        
        JOptionPane.showMessageDialog(
                this,
                "Dados:\n" + pessoa.toString(),
                "Relat√≥rio",
                JOptionPane.PLAIN_MESSAGE);
    }//fecha cadastrar
    
    private void limpar() {
        jtNome.setText("");
        jtNome.grabFocus();
        
//        jrMasculino.setSelected(false);
//        jrFeminino.setSelected(false);

        bgSexo.clearSelection();
    }//fecha limpar
    
    private void jbCadastrarActionPerformed(java.awt.event.ActionEvent evt) {                                            
        cadastrar();
        limpar();
    }                                           

    private void jbLimparActionPerformed(java.awt.event.ActionEvent evt) {                                         
        limpar();
    }

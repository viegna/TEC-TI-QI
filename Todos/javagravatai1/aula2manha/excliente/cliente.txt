public class Cliente { //forma de bolo
    
    private String nome;
    private String sexo;
    
    public Cliente() {
    
    
    public Cliente(String nome, String sexo) {
        this.nome = nome;
        this.sexo = sexo;
    }
    
    public String getNome() {
        return nome;
    }
    
    public void setNome(String nome) {
        this.nome = nome;
    }
    
    public String getSexo() {
        return sexo;
    }
    
    public void setSexo(String sexo) {
        this.sexo = sexo;
    }
    
    public String toString() {
        return "\nNome " + nome
             + "\nSexo " + sexo;
    }
}//fecha cliente

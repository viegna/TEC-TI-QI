    private void cadastrar() {
        Pessoa pessoa = new Pessoa();
        
        pessoa.setNome(jtNome.getText());
        pessoa.setPeso(Double.parseDouble(jtPeso.getText()));
        pessoa.setIdade(Integer.parseInt(jtIdade.getText()));
        
        JOptionPane.showMessageDialog(
                rootPane,
                "Relat√≥rio:" + pessoa.toString(),
                "Dados",
                JOptionPane.PLAIN_MESSAGE);
        
    }//fecha cadastrar    

    private void limpar() {
        jtNome.setText("");
        jtPeso.setText("");
        jtIdade.setText("");
        
        jtNome.grabFocus();
    }//fecha limpar
    
    private void jbCadastrarActionPerformed(java.awt.event.ActionEvent evt) {                                            
        cadastrar();
        limpar();
    }                                           

    private void jbLimparActionPerformed(java.awt.event.ActionEvent evt) {                                         
        limpar();
    }                                        

    private void jbCadastrarKeyPressed(java.awt.event.KeyEvent evt) {                                       
        if(evt.getKeyCode() == KeyEvent.VK_ENTER) {
            cadastrar();
            limpar();
        }//fecha if
    }                                      

    private void jbLimparKeyPressed(java.awt.event.KeyEvent evt) {

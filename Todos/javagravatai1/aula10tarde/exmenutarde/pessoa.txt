package br.edu.qi.modelo;

/**
 *
 * @author Alunos
 */
public class Pessoa {
    
    private String nome;
    private int idade;

    public Pessoa() {
    }

    public Pessoa(String nome, int idade) {
        this.nome = nome;
        this.idade = idade;
    }

    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public int getIdade() {
        return idade;
    }

    public void setIdade(int idade) {
        this.idade = idade;
    }

    public int calcularIdade() throws Exception {
     int resposta = 0;
        try {
            resposta = idade * 12;
        } catch (Exception exception) {
            throw new Exception("Erro!");
        }
        return resposta;
    }
    
    @Override
    public String toString() {
        return "\nNome = " + nome 
             + "\nIdade = " + idade;
    }
}

    private void cadastrar() {
        try {
            Pessoa pessoa = new Pessoa();
            pessoa.setNome(jtNome.getText());
            pessoa.setIdade(
                Integer.parseInt(
                    jtIdade.getText()));
            JOptionPane.showMessageDialog(
                this, 
                "\nDados: " + pessoa.toString()
               +"\nidade em meses:"+pessoa.calcularIdade());
        } catch (Exception e) {
            JOptionPane.showMessageDialog(
                    this, 
                    "Erro! " + e.getMessage());
        }//fecha catch
    }//fecha cadastrar
    
    private void limpar() {
        jtNome.setText("");
        jtIdade.setText("");
        jtNome.grabFocus();
    }
    
    private void jbCadastrarActionPerformed(java.awt.event.ActionEvent evt) {                                            
        cadastrar();
        limpar();
    }
